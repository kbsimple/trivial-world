    1. Create apps/generator/lib/ollama/verification.ts:
       - Import generateText from 'ai'
       - Import createOllama from 'ollama-ai-provider-v2'
       - Import Question from '@trivial-world/types'
       - Import DEFAULT_MODEL, getOllamaClient from './client'

       - Define VERIFICATION_PROMPTS object with 3 prompts:
         1. factualAccuracy: "Verify this trivia question is factually correct. Question: '{questionText}' Answer: '{answerText}'. Answer only 'correct' or 'incorrect' with a brief explanation."
         2. alternatePhrasing: "Is the following statement true or false? {questionText} The answer is {answerText}. Provide your reasoning."
         3. reverseVerification: "If someone told you '{answerText}' is the answer to '{questionText}', would they be correct? Verify this claim independently. Answer yes/no with reasoning."

       - Export interface VerificationResult { pass: number, prompt: string, response: string, passed: boolean }
       - Export interface ConfidenceScore { score: number (0-100), passes: number (0-3), results: VerificationResult[], needsReview: boolean }

       - Export async function verifyQuestion(question: Question, model?: string, ollamaUrl?: string): Promise<ConfidenceScore>:
         - Get Ollama client with getOllamaClient(ollamaUrl)
         - Sequentially call generateText for each verification prompt
         - Parse response for pass/fail using evaluatePassResult (look for 'correct', 'true', 'yes', 'accurate' keywords)
         - Calculate passes = results.filter(r => r.passed).length
         - Calculate score = Math.round((passes / 3) * 100)
         - Set needsReview = passes < 3 per D-10
         - Return ConfidenceScore object

       - Export function evaluatePassResult(response: string): boolean:
         - Lowercase response
         - Return true if contains 'correct', 'true', 'yes', or ('accurate' without 'not')
         - Return false otherwise

    Per D-07: Multi-pass verification with 3 different phrasings.
    Per D-08: Confidence scoring based on pass agreement (3/3=100%, 2/3=67%, 1/3 or 0/3=flagged).
    Per D-10: Questions marked 'needs review' when passes disagree.
    Per RESEARCH.md: Sequential verification to catch hallucinations.
