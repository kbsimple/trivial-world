export const QuestionSchema = z.object({
  id: z.string().regex(/^[a-z0-9-]+$/),
  category: CategorySchema,
  questionText: z.string().min(10).max(500),
  answerText: z.string().min(1).max(200),
  difficulty: DifficultySchema.optional(),
  choices: z.array(z.string()).max(6).optional(),
  correctChoiceIndex: z.number().int().min(0).optional(),
  createdAt: z.string().datetime().optional(),
  updatedAt: z.string().datetime().optional(),
  author: z.string().max(100).optional(),
  source: z.string().url().optional(),
});
export type Question = z.infer<typeof QuestionSchema>;

## Trust Boundaries

| Boundary | Description |
|----------|-------------|
| Browser → Ollama | Untrusted user input crosses to local LLM service |
| User Input → LLM Prompt | Topic/guidance text injected into generation prompt |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-07-01 | Tampering | Topic input field | mitigate | sanitizeInput strips control chars, limits to 100 chars per RESEARCH.md Code Examples |
| T-07-02 | Tampering | Guidance textarea | mitigate | sanitizeInput strips control chars, limits to 500 chars |
| T-07-03 | Information Disclosure | Ollama endpoint | accept | Local network only, no secrets transmitted |
| T-07-04 | Tampering | Category selection | mitigate | Validate category against CategorySchema enum before LLM call |
| T-07-05 | Denial of Service | Generation batch size | mitigate | Limit to 10 questions per batch per D-16 to prevent excessive LLM calls |
</threat_model>
