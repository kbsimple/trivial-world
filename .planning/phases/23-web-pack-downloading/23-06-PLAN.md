/**
 * Unit tests for packCache.web.ts (idb-keyval IDB adapter).
 *
 * Import packCache.web.ts directly — Vitest does not resolve .web.ts extensions
 * the way Metro does. vi.mock('idb-keyval') replaces the IDB calls.
 */
import { describe, it, expect, vi, beforeEach } from 'vitest';

// Mock idb-keyval before importing the module under test
vi.mock('idb-keyval', () => {
  const store = new Map<string, unknown>();
  return {
    createStore: vi.fn(() => 'mock-store'),
    get: vi.fn(async (key: string) => store.get(key)),
    set: vi.fn(async (key: string, value: unknown) => { store.set(key, value); }),
    del: vi.fn(async (key: string) => { store.delete(key); }),
    keys: vi.fn(async () => [...store.keys()]),
    __store: store,  // exposed for test cleanup
  };
});

import {
  getCachedPackQuestions,
  setCachedPackQuestions,
  getCachedPackChecksum,
  setCachedPackChecksum,
  getCachedPackIndex,
  setCachedPackIndex,
  getOfflinePackIds,
  deleteCachedPack,
  requestPersistentStorage,
} from './packCache.web';
import type { Question, PackIndexEntry } from '@trivial-world/types';

// Helper to get the mock store for cleanup between tests
const getMockStore = async () => {
  const { __store } = await import('idb-keyval') as any;
  return __store as Map<string, unknown>;
};

beforeEach(async () => {
  const store = await getMockStore();
  store.clear();
});

const mockQuestion: Question = {
  id: 'q1',
  category: 'blue' as any,
  questionText: 'Test question?',
  answerText: 'Test answer',
};

const mockEntry: PackIndexEntry = {
  id: 'pack-1',
  name: 'Test Pack',
  description: 'A test pack',
  version: '1.0.0',
  downloadUrl: 'https://example.com/pack-1.json',
  checksum: 'abc123',
  size: 1000,
  categoryCounts: {},
  totalQuestions: 1,
} as any;

describe('getCachedPackQuestions', () => {
  it('returns null when pack is not in IDB', async () => {
    const result = await getCachedPackQuestions('pack-1');
    expect(result).toBeNull();
  });

  it('returns Question[] after setCachedPackQuestions', async () => {
    await setCachedPackQuestions('pack-1', [mockQuestion]);
    const result = await getCachedPackQuestions('pack-1');
    expect(result).toEqual([mockQuestion]);
  });
});

describe('getCachedPackChecksum', () => {
  it('returns null when checksum not stored', async () => {
    expect(await getCachedPackChecksum('pack-1')).toBeNull();
  });

  it('returns stored checksum string', async () => {
    await setCachedPackChecksum('pack-1', 'sha256-abc');
    expect(await getCachedPackChecksum('pack-1')).toBe('sha256-abc');
  });
});

describe('getCachedPackIndex', () => {
  it('returns null when index not stored', async () => {
    expect(await getCachedPackIndex()).toBeNull();
  });

  it('returns stored PackIndexEntry[] after setCachedPackIndex', async () => {
    await setCachedPackIndex([mockEntry]);
    expect(await getCachedPackIndex()).toEqual([mockEntry]);
  });
});

describe('getOfflinePackIds', () => {
  it('returns empty array when no packs cached', async () => {
    expect(await getOfflinePackIds()).toEqual([]);
  });

  it('returns packIds for stored packs', async () => {
    await setCachedPackQuestions('pack-1', [mockQuestion]);
    await setCachedPackQuestions('pack-2', [mockQuestion]);
    const ids = await getOfflinePackIds();
    expect(ids.sort()).toEqual(['pack-1', 'pack-2']);
  });

  it('excludes pack-checksum keys', async () => {
    await setCachedPackQuestions('pack-1', [mockQuestion]);
    await setCachedPackChecksum('pack-1', 'abc');
    const ids = await getOfflinePackIds();
    expect(ids).toEqual(['pack-1']);
    expect(ids).not.toContain('pack-checksum:pack-1');
  });
});

describe('deleteCachedPack', () => {
  it('removes pack questions AND checksum', async () => {
    await setCachedPackQuestions('pack-1', [mockQuestion]);
    await setCachedPackChecksum('pack-1', 'abc');
    await deleteCachedPack('pack-1');
    expect(await getCachedPackQuestions('pack-1')).toBeNull();
    expect(await getCachedPackChecksum('pack-1')).toBeNull();
  });
});

describe('requestPersistentStorage', () => {
  it('returns false when navigator.storage.persist is not available', async () => {
    // jsdom does not implement navigator.storage.persist
    const result = await requestPersistentStorage();
    expect(typeof result).toBe('boolean');
  });
});
```

describe('downloadPackForOffline', () => {
  const entry: PackIndexEntry = {
    id: 'pack-1',
    name: 'Test Pack',
    description: 'Desc',
    version: '1.0.0',
    downloadUrl: 'https://example.com/pack-1.json',
    checksum: 'new-checksum',
    size: 500,
    categoryCounts: {},
    totalQuestions: 1,
  } as any;

  beforeEach(() => {
    // Reset store state
    usePackStore.setState({
      isDownloading: false,
      downloadProgress: 0,
      downloadBytesWritten: 0,
      downloadError: null,
      offlinePackIds: [],
    });
  });

  it('skips re-download when stored checksum matches entry.checksum', async () => {
    const { getCachedPackChecksum } = await import('../services/packCache.web');
    vi.mocked(getCachedPackChecksum).mockResolvedValueOnce('new-checksum');

    await usePackStore.getState().downloadPackForOffline({ ...entry, checksum: 'new-checksum' });

    expect(usePackStore.getState().isDownloading).toBe(false);
    expect(usePackStore.getState().downloadProgress).toBe(100);
    // fetch should NOT have been called
    expect(mockFetch).not.toHaveBeenCalled();
  });

  it('downloads, validates, and stores pack when checksum differs', async () => {
    const { setCachedPackQuestions, setCachedPackChecksum } = await import('../services/packCache.web');
    const mockQuestions = [{ id: 'q1', category: 'blue', questionText: 'Q?', answerText: 'A' }];
    const mockPackBody = JSON.stringify({
      metadata: {
        id: 'pack-1', name: 'Test Pack', version: '1.0.0',
        description: '', author: '', totalQuestions: 1,
        categoryCounts: {}, schemaVersion: 1,
      },
      questions: mockQuestions,
    });

    // Mock ReadableStream response
    const encoder = new TextEncoder();
    const bytes = encoder.encode(mockPackBody);
    mockFetch.mockResolvedValueOnce({
      ok: true,
      body: {
        getReader: () => {
          let done = false;
          return {
            read: vi.fn().mockImplementation(async () => {
              if (!done) { done = true; return { done: false, value: bytes }; }
              return { done: true, value: undefined };
            }),
          };
        },
      },
    });

    await usePackStore.getState().downloadPackForOffline(entry);

    expect(vi.mocked(setCachedPackQuestions)).toHaveBeenCalledWith('pack-1', expect.any(Array));
    expect(vi.mocked(setCachedPackChecksum)).toHaveBeenCalledWith('pack-1', 'new-checksum');
    expect(usePackStore.getState().offlinePackIds).toEqual(['pack-1']);
    expect(usePackStore.getState().isDownloading).toBe(false);
  });

  it('sets downloadError and re-throws on fetch failure', async () => {
    mockFetch.mockRejectedValueOnce(new Error('Network error'));

    await expect(
      usePackStore.getState().downloadPackForOffline(entry)
    ).rejects.toThrow('Network error');

    expect(usePackStore.getState().downloadError).toBe('Network error');
    expect(usePackStore.getState().isDownloading).toBe(false);
  });
});
```
